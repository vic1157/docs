## Scroll System

### Scroll Interface

```typescript
interface Scroll {
  id: string;
  title: string;
  scriptureReference: string;
  duration: string;
  timestamp: string;
  hasQuiz: boolean;
  quizScore?: { correct: number; total: number };
  summary: string;
  questions: Question[];
  reflection: string;
  readingTime: string;
}
```

<ResponseField name="id" type="string" required>
  Unique identifier for the scroll
</ResponseField>

<ResponseField name="title" type="string" required>
  Display title for the scroll
</ResponseField>

<ResponseField name="scriptureReference" type="string" required>
  Bible passage reference (e.g., "Jeremiah 29:10-14")
</ResponseField>

<ResponseField name="hasQuiz" type="boolean" required>
  Whether a quiz exists for this scroll
</ResponseField>

<ResponseField name="quizScore" type="object">
  Optional quiz score with `correct` and `total` properties
</ResponseField>

### Question Interface

```typescript
interface Question {
  id: string;
  type: 'understanding' | 'discussion';
  text: string;
  verseReference?: string;
}
```

### QuizQuestion Interface

```typescript
interface QuizQuestion {
  id: string;
  question: string;
  options: string[];
  correctAnswer: number;
  explanation?: string;
  verseReference?: string;
}
```

<ResponseField name="correctAnswer" type="number" required>
  Zero-based index of the correct option
</ResponseField>
