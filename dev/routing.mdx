## URL Parameter Patterns

### Onboarding Trigger

```typescript
// Navigate with parameter to trigger onboarding modal
navigate("/profile?onboarding=true");

// Check parameter in component
const searchParams = new URLSearchParams(location.search);
const isOnboarding = searchParams.get("onboarding") === "true";

if (isOnboarding) {
  setShowOnboardingModal(true);
}
```

### Auto-start Quiz

```typescript
// Navigate to scroll with auto-start quiz
navigate(\`/scroll/\${scrollId}?startQuiz=true\`);

// Check parameter in ScrollDetail component
const [searchParams] = useSearchParams();
const shouldStartQuiz = searchParams.get("startQuiz") === "true";
```

### Chat ID Parameter

```typescript
// Load specific chat
navigate(\`/chat?id=\${chatId}\`);

// Access in Chat component
const [searchParams] = useSearchParams();
const chatId = searchParams.get("id");
```

## Navigation State

### Passing State Between Routes

```typescript
// Navigate with state
navigate("/home", {
  state: { dailyScrollCompleted: true }
});

// Access state in destination
const location = useLocation();
const { dailyScrollCompleted } = location.state || {};
```

## Router Setup

```typescript App.tsx
import { BrowserRouter, Routes, Route } from "react-router-dom";

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <ThemeProvider>
        <BrowserRouter>
          <Routes>
            <Route path="/" element={<Loading />} />
            <Route path="/login" element={<Login />} />
            {/* ... 29 more routes */}
            <Route path="*" element={<NotFound />} />
          </Routes>
        </BrowserRouter>
      </ThemeProvider>
    </QueryClientProvider>
  );
}
```

## Navigation Helpers

### useNavigate Hook

```typescript
import { useNavigate } from "react-router-dom";

const navigate = useNavigate();

// Simple navigation
navigate("/home");

// With state
navigate("/home", { state: { fromOnboarding: true } });

// With replace (no history entry)
navigate("/home", { replace: true });
```

### Link Component

```tsx
import { Link } from "react-router-dom";

<Link to="/profile" className="...">
  Go to Profile
</Link>
```
